"Sin respuesta de GPT."; return new Response(JSON.stringify({ result: reply }), { headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*", }, }); } catch (err) { return new Response( JSON.stringify({ error: "Fallo en la función.", detalle: err.message ?? err.toString(), }), { status: 500, headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*", }, } ); } });
